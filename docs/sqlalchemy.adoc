        FruitsMenu
    ).filter(
        sqlalchemy.or_(
            FruitsMenu.name == 'Apple',
            FruitsMenu.name == 'Orange'
        )
    ).with_entities(
        FruitsMenu.name,
        FruitsMenu.price

    items = query_obj.all()

    records = []

        compile_kwargs={"literal_binds": True}
    )
